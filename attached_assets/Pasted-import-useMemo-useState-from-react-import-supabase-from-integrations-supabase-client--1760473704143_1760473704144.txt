import { useMemo, useState } from "react";
import { supabase } from "@/integrations/supabase/client";
import { Badge } from "@/components/ui/badge";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";
import { Label } from "@/components/ui/label";

type Target = 'ADD_QUANTITY' | 'FILL_TO_TABS' | 'FILL_TO_TABS_PLUS' | 'FILL_TO_FULL';

export function FuelOrderForm({
  aircraft,
  latestFuel, // { gallons_onboard: number } from v_aircraft_fuel_latest (0 if none)
}: {
  aircraft: { id: string; tail: string; fuel_type: string; usable_fuel_gal: number; tabs_fuel_gal: number };
  latestFuel: { gallons_onboard: number } | null;
}) {
  const [target, setTarget] = useState<Target>('ADD_QUANTITY');
  const [addQty, setAddQty] = useState<number | ''>('');
  const [tabsPlus, setTabsPlus] = useState<number | ''>('');
  const current = latestFuel?.gallons_onboard ?? 0;
  const full = aircraft.usable_fuel_gal ?? 0;
  const tabs = aircraft.tabs_fuel_gal ?? 0;

  const previewAdd = useMemo(() => {
    if (full <= 0) return 0;
    const safe = (x: number) => Math.max(0, Math.min(x, full));
    if (target === 'ADD_QUANTITY') {
      const q = typeof addQty === 'number' ? addQty : 0;
      return safe(Math.min(q, full - current));
    }
    if (target === 'FILL_TO_TABS') {
      return safe(Math.max(tabs - current, 0));
    }
    if (target === 'FILL_TO_TABS_PLUS') {
      const plus = typeof tabsPlus === 'number' ? tabsPlus : 0;
      const goal = Math.min(tabs + plus, full);
      return safe(Math.max(goal - current, 0));
    }
    if (target === 'FILL_TO_FULL') {
      return safe(Math.max(full - current, 0));
    }
    return 0;
  }, [target, addQty, tabsPlus, current, full, tabs]);

  async function submitOrder(e: React.FormEvent) {
    e.preventDefault();
    const payload: any = {
      aircraft_id: aircraft.id,
      target,
      add_quantity_gal: target === 'ADD_QUANTITY' ? (addQty || null) : null,
      tabs_plus_gal: target === 'FILL_TO_TABS_PLUS' ? (tabsPlus || null) : null,
      notes: null,
    };
    const { error } = await supabase.from('fuel_orders').insert(payload);
    if (error) {
      console.error(error);
      // toast error
      return;
    }
    // Option A: if fueling is immediate in your flow, also insert fuel_logs right now using previewAdd
    // const { error: e2 } = await supabase.from('fuel_logs').insert({ aircraft_id: aircraft.id, quantity: previewAdd });
    // if (!e2) the trigger will update the snapshot
    // toast success / invalidate queries
  }

  return (
    <form onSubmit={submitOrder} className="space-y-4">
      <div className="flex items-center justify-between">
        <div className="text-sm">
          Aircraft <span className="font-medium">{aircraft.tail}</span>
        </div>
        <div className="flex gap-2">
          <Badge variant="secondary">Fuel: {aircraft.fuel_type.replaceAll('_',' ')}</Badge>
          <Badge variant="outline">Current: {current.toFixed(1)} gal</Badge>
          <Badge variant="outline">Tabs: {tabs.toFixed(1)} gal</Badge>
          <Badge variant="outline">Full: {full.toFixed(1)} gal</Badge>
        </div>
      </div>

      <RadioGroup value={target} onValueChange={(v) => setTarget(v as Target)} className="grid gap-3">
        <div className="flex items-center space-x-2">
          <RadioGroupItem value="ADD_QUANTITY" id="addqty" />
          <Label htmlFor="addqty" className="flex-1">Add specific gallons</Label>
          <Input
            className="w-32"
            type="number"
            step="0.1"
            min={0}
            disabled={target !== 'ADD_QUANTITY'}
            value={addQty}
            onChange={(e) => setAddQty(e.target.value === '' ? '' : Number(e.target.value))}
            placeholder="e.g. 20.0"
          />
        </div>

        <div className="flex items-center space-x-2">
          <RadioGroupItem value="FILL_TO_TABS" id="filltabs" />
          <Label htmlFor="filltabs" className="flex-1">Fill to Tabs</Label>
        </div>

        <div className="flex items-center space-x-2">
          <RadioGroupItem value="FILL_TO_TABS_PLUS" id="filltabsplus" />
          <Label htmlFor="filltabsplus" className="flex-1">Fill to Tabs +</Label>
          <Input
            className="w-32"
            type="number"
            step="0.1"
            min={0}
            disabled={target !== 'FILL_TO_TABS_PLUS'}
            value={tabsPlus}
            onChange={(e) => setTabsPlus(e.target.value === '' ? '' : Number(e.target.value))}
            placeholder="e.g. 10.0"
          />
        </div>

        <div className="flex items-center space-x-2">
          <RadioGroupItem value="FILL_TO_FULL" id="fillfull" />
          <Label htmlFor="fillfull" className="flex-1">Fill to Full</Label>
        </div>
      </RadioGroup>

      <div className="text-sm text-muted-foreground">
        Computed Add: <span className="font-semibold">{previewAdd.toFixed(1)} gal</span>
      </div>

      <Button type="submit">Request Fuel</Button>
    </form>
  );
}
